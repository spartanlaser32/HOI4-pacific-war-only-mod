			limit = {

					has_dynamic_modifier = { modifier = PHI_asian_trade_connections_dynamic_modifier }

				limit = {
					FROM = {

					}

			else_if = {
				limit = {
					FROM = {
						original_tag = MAL
					}

			}
			else = {
				custom_effect_tooltip = idea_desc|PHI_asian_trade_connections_dutch_east_india
			}
		}
		if = {
			limit = {
				FROM = { TAG = SIA }
			}
			hidden_effect = { 
				add_ideas = PHI_asian_trade_connections_siam
			}
		}
		else_if = {
			limit = {
				FROM = { TAG = INS }
