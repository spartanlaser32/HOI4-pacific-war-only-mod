country_event = { #INVASION OF POLAND
	id = pwo_world_events.5
	title =  pwo_world_events.5.t
	desc = pwo_world_events.5.d

	is_triggered_only = yes

	major = yes

	immediate = {
		add_named_threat = { threat = 5 name = GER_danzig_or_war }
		add_named_threat = { threat = 2 name = GER_anti_comintern_pact }
		set_global_flag = GER_ENG_WAR
		ENG = {
			add_ideas = { PWO_major_at_war }
		}
		FRA = {
			add_ideas = { PWO_major_at_war }
		}
		RAJ = {
			add_ideas = { PWO_minor_at_war }
		}
		CAN = {
			add_ideas = { PWO_minor_at_war }
		}
		AST = {
			add_ideas = { PWO_minor_at_war }
		}
		NZL = {
			add_ideas = { PWO_minor_at_war }
		}
		MAL = {
			add_ideas = { PWO_minor_at_war }
		}
		USA = { puppet = ENG }
		USA = {
			set_autonomy = {
				target = ENG
				autonomy_state = autonomy_us_ally
			}
		}
	}

	option = {
		name = pwo_world_events.5.a
	}
}
